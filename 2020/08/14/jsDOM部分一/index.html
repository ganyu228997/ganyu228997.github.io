js第二部分DOM部分一

无标签
发布日期:   2020-08-14

获取元素

1.根据ID获取元素：document.getElementById()
2.根据标签名获取：document.getElementsByTagName()
3.通过HTML5新增方法获取元素：
（1） 通过类名获取元素：document.getElementsByClassName();
（2）querySelector返回指定选择器的第一个元素对象：document.querySelector()
（3）querySelectorAll返回指定选择器的所有元素对象：querySelectorAll()
4.获取body 元素：document.body
5.获取html元素：document.documentElement

var btn = document.getElementById(‘btn’);

alert(‘点秋香’);

tab栏切换

主要利用排他思想、自定义属性（setAttribute(‘属性名’,属性值)）和获取自定义属性（getAttribute(‘属性名’)）。

    var tab_list = document.querySelector('.tab_list');
    var lis = tab_list.querySelectorAll('li');
    var itm = document.querySelectorAll('.itm');
    // 给每个li添加一个自定义属性
    for (var i=0; i<lis.length;i++) {
        lis[i].setAttribute('index',i)
        lis[i].onclick = function() {
            for (var i=0; i<lis.length;i++) {
                lis[i].className = '';
            }
            this.className = 'current';
            // 获取自定义属性
            var index = this.getAttribute('index');
            // 隐藏所有tab栏对应的内容
            for(i = 0;i <itm.length;i++) {
                itm[i].style.display = 'none'
            }
            // 显示当前点击的tab栏的内容
            itm[index].style.display = 'block';
        }
    }

节点操作

1.父节点: 目标节点.parentNode
2.子节点: 目标节点.childNodes
3 获取第一个和最后一个子节点
（1）利用first/lastchild获取子节点，可能包含其它类型节点
console.log(ol.firstChild);
console.log(ol.lastChild);
（2）利用first/lastElementChild 只支持IE9以上
console.log(ol.firstElementChild);
console.log(ol.lastElementChild);
（3）利用 children[n] 获取任意一个节点
第一个
console.log(ol.children[0]);
最后一个
console.log(ol.children[ol.children.length - 1]);
4.兄弟节点
var div = document.querySelector(‘div’);
一、（1）nextSibling获取div的下一个兄弟节点：包含文本节点元素节点等
console.log(div.nextSibling);
（2）previousSibling获取div的上一个兄弟节点: 包含文本节点元素节点等
console.log(div.previousSibling);
二、（1）nextElementSibling获取div的下一个元素节点
console.log(div.nextElementSibling);
（2）上一个元素节点
console.log(div.previousElementSibling);
5.创建并添加节点
(1).创建节点元素
var li = document.createElement(‘li’);
(2).添加节点
一 、依次往后添加:node.appendChild（child） node父级 child子级
ul.appendChild(li)
二、添加到指定元素前面:node.insertBefore（child，指定元素）
var lili = document.createElement(‘li’);
ul.insertBefore(lili,ul.children[0]);
6.删除节点: 父节点.removeChild(目标节点)
ul.removeChild(ul.children[0]) 删除ul里面的第一个节点
7.克隆节点: node.cloneNode（）括号里为空或者false为浅拷贝只复制标签不复制里面的内容；括号里为true为深拷贝复制标签及里面的内容
var lili = ul.children[0].cloneNode(true);

    var btn = document.querySelector('button');
    var text = document.querySelector('textarea')
    var ul = document.querySelector('ul');
    btn.onclick = function () {
        var li = document.createElement('li');
        li.innerHTML = text.value + "<a href='javascript:;'>删除</a>";
        if (text.value == '') {
            alert('您输入的内容为空');
        }
        else {
            // 加到后面
            // ul.appendChild(li);

            ul.insertBefore(li, ul.children[0]);
            var as = document.querySelectorAll('a');
            for (var i = 0; i < as.length; i++) {
                as[i].onclick = function () {
                    ul.removeChild(this.parentNode);
                }
            }

        }
    }

动态生成表格

创建元素的三种方式和区别

1.document.write()创建（了解） 是直接将内容写入页面的内容流，但是文档流执行完毕，则它会导致页面全部重绘
2.innerHTML创建元素不要采用字符串拼接形式，采取数组的形式效率最高

var inner = document.querySelector('.inner');
var arr = [];
for(var i = 0; i <= 100; i++) {
    arr.push('<a href="#">百度</a>')
}
inner.innerHTML = arr.join('');

3.document.createElement（）创建元素 效率稍微低一点，但结构更清晰

 var create = document.querySelector('.create');
for(var i = 0; i <= 100; i++) {
    var a = document.createElement('a');
    create.appendChild(a);
}
